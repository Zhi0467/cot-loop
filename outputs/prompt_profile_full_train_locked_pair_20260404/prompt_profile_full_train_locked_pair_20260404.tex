\section*{Definitions}
\subsection*{Targets}
\begin{itemize}[leftmargin=1.5em]
\item \texttt{mean\_relative\_length}: for prompt \texttt{i}, this is the repeated-rollout mean of \texttt{L\_r / E}, where \texttt{L\_r} is rollout length and \texttt{E} is the prompt's effective token budget.
\item \texttt{majority\_s\_0.5}: for prompt \texttt{i}, this is \texttt{1} if a strict majority of repeated rollouts satisfy \texttt{L\_r / E >= 0.5}, otherwise \texttt{0}.
\end{itemize}

\subsection*{Views}
\begin{itemize}[leftmargin=1.5em]
\item \texttt{ensemble} for regression: one MLP per layer with \texttt{mean\_prob} aggregation.
\item \texttt{ensemble} for binary: one MLP per layer with \texttt{vote\_fraction} aggregation over per-layer \texttt{0/1} predictions.
\item \texttt{last\_layer}: the same probe family restricted to the final layer only.
\end{itemize}

\subsection*{Evaluation Pairs}
\begin{itemize}[leftmargin=1.5em]
\item Regression eval pair: each held-out prompt \texttt{i} contributes one aligned pair \texttt{(\^{}y\_i, y\_i)}, where \texttt{y\_i} is that same prompt's realized \texttt{mean\_relative\_length}.
\item Binary eval pair: each held-out prompt \texttt{i} contributes one aligned pair \texttt{(\^{}s\_i, y\_i)}, where \texttt{y\_i} is that same prompt's realized \texttt{majority\_s\_0.5} label.
\end{itemize}

\subsection*{Metrics}
\begin{itemize}[leftmargin=1.5em]
\item \texttt{top\_10p\_capture} / \texttt{top\_20p\_capture}: sort held-out prompts by predicted score, keep the top \texttt{10\%} or \texttt{20\%}, and ask what fraction of total target mass falls inside that bucket. For \texttt{mean\_relative\_length}, target mass means summed realized relative-length mass over held-out prompts.
\item \texttt{RMSE}: pointwise error on the aligned held-out prompt pairs.
\item \texttt{Spearman}: monotone ordering agreement on those same aligned held-out prompt pairs. It is not the task definition.
\item \texttt{PR-AUC}: ranking metric for the binary head across thresholds.
\end{itemize}

\subsection*{Metadata Baseline}
\begin{itemize}[leftmargin=1.5em]
\item Allowed inputs: \texttt{prompt\_token\_count}, and separately \texttt{effective\_max\_tokens}.
\item Excluded inputs: no activations, no prompt text, no dataset/source identity, no model identifier, no architecture features, no generated output features.
\item Regression baseline: standardized linear regression fit on the train split and evaluated once on held-out test.
\item Binary baseline: train-fit 1D score rule, with direction chosen by train \texttt{PR-AUC} and threshold chosen by train macro-F1 / positive-F1 / accuracy, evaluated once on held-out test.
\item Important fixed-run nuance: \texttt{effective\_max\_tokens=30000} is constant on this run, so prompt length is the only nontrivial metadata feature here. The joint regression control collapses to prompt length, and the binary \texttt{effective\_budget} control is effectively constant.
\end{itemize}

\subsection*{Regression Interpretation}

  \begin{itemize}[leftmargin=1.5em]
  \item wins: \texttt{GPQA}, \texttt{LiveCodeBench}
  \item losses: \texttt{AIME}, \texttt{MATH-500}, \texttt{MMLU-Pro}
  \end{itemize}
\item Ensemble versus \texttt{last\_layer}:
  \begin{itemize}[leftmargin=1.5em]
  \item on \texttt{top\_20p\_capture}, ensemble wins \texttt{4 / 5}; only \texttt{AIME} slightly favors \texttt{last\_layer}
  \item on \texttt{RMSE}, ensemble also wins \texttt{4 / 5}, but the exception is different: \texttt{LiveCodeBench}
  \end{itemize}
\item Most important split: \texttt{LiveCodeBench} flips depending on the object. \texttt{last\_layer} is slightly better calibrated by \texttt{RMSE}, but ensemble is better for screening by both \texttt{top\_10p\_capture} and \texttt{top\_20p\_capture}.
\item Practical read: \texttt{mean\_relative\_length} is still a usable score, but it is not a uniform activation-lift story over prompt length. It is mixed even after freezing the target and the run contract.
\end{itemize}

\subsection*{Binary Interpretation}
\begin{itemize}[leftmargin=1.5em]
\item Ensemble \texttt{PR-AUC} beats prompt length on all five datasets.
\item The cleanest wins are:
  \begin{itemize}[leftmargin=1.5em]
  \item \texttt{AIME}: strong ranking and strong fixed-threshold behavior
  \item \texttt{MMLU-Pro}: large ranking gain despite tiny prevalence
  \item \texttt{LiveCodeBench}: large ranking gain with solid precision/recall
  \end{itemize}
\item Caveats by dataset:
  \begin{itemize}[leftmargin=1.5em]
  \item \texttt{GPQA}: real ranking lift, but only \texttt{5\%} test prevalence, so threshold metrics are unstable
  \item \texttt{MATH-500}: ranking lift over prompt length exists, but the absolute threshold-quality story is still weak
  \end{itemize}
\end{itemize}

\section*{What This Run Does And Does Not Show}
\begin{itemize}[leftmargin=1.5em]
\item This run does show:
  \begin{itemize}[leftmargin=1.5em]
  \item the locked pair can be trained end-to-end on the full saved surface under one frozen contract
  \item \texttt{majority\_s\_0.5} is the cleaner finished activation-lift head today
  \item \texttt{mean\_relative\_length} is usable as a screening score, but only with a mixed cross-dataset read
  \end{itemize}
\item This run does not show:
  \begin{itemize}[leftmargin=1.5em]
  \item that \texttt{Spearman} should be the primary regression metric
  \item that \texttt{mean\_relative\_length} beats prompt length uniformly
  \item that \texttt{majority\_s\_0.5} is a pure loop label
  \item that \texttt{p\_loop} should be reopened as the default train target after the fact
  \end{itemize}
\end{itemize}

\section*{Recommendation}
\begin{itemize}[leftmargin=1.5em]
\item Default finished surface from this run: \texttt{majority\_s\_0.5}
\item How to report the continuous head:
  \begin{itemize}[leftmargin=1.5em]
  \item if the use is screening high-risk prompts, report \texttt{top\_20p\_capture} first
  \item use \texttt{top\_10p\_capture} second if tighter budget matters
  \item keep \texttt{RMSE} as calibration context
  \item keep \texttt{Spearman} only as a tertiary diagnostic
  \end{itemize}
\item Most honest one-line summary: on the frozen locked run, the binary head is a \texttt{5 / 5} prompt-length-baseline win by \texttt{PR-AUC}, while the continuous head is a \texttt{3 / 5} win by screening capture and only a \texttt{2 / 5} win by calibration error.
\end{itemize}
